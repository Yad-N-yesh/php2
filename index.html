        echo "<br>Intersection:<br>";

            return $num%2==0;
        });

        break;

        echo "<br>Sorted names:<br>";

<?php

function swapbyvalue($a,$b)
{
    echo "First number:$a and second is:$b<br>";
    $s=$a;
    $a=$b;
    $b=$s;

    echo "First value:$a and second is:$b<br><br>";
}

function swapbyreference(&$x,&$y)
{
    echo "First number:$x and second is:$y<br>";
    $s=$x;
    $x=$y;
    $y=$s;

    echo "First value:$x and second is:$y";
}

$m=10;
$n=20;

swapbyvalue($m,$n);
swapbyreference($m,$n);

?>
